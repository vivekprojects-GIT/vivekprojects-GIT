How to integrate the code review API into your workflow

import requests

# Review code using the API
def review_code(code, language="python"):
    url = "https://api.code-review.ai/v1/review"
    payload = {
        "code": code,
        "language": language,
        "options": {
            "security_scan": True,
            "performance_analysis": True,
            "best_practices": True
        }
    }
    
    response = requests.post(url, json=payload)
    return response.json()

# Example usage
code = """
def process_user_input(user_input):
    query = "SELECT * FROM users WHERE name = '" + user_input + "'"
    return execute_query(query)
"""

result = review_code(code)
print(f"Found {len(result['issues'])} issues")
for issue in result['issues']:
    print(f"- {issue['severity']}: {issue['message']}")

async function reviewCode(code, language = 'javascript') {
  const response = await fetch('https://api.code-review.ai/v1/review', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer YOUR_API_KEY'
    },
    body: JSON.stringify({
      code: code,
      language: language,
      options: {
        security_scan: true,
        performance_analysis: true,
        best_practices: true
      }
    })
  });
  
  return await response.json();
}

// Example usage
const code = `
function processUserInput(userInput) {
  const query = "SELECT * FROM users WHERE name = '" + userInput + "'";

}
`;

  });
});
